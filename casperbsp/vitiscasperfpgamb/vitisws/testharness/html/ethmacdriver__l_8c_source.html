1270 u64 XGMAC_ReadARPCacheEntry(UINTPTR BaseAddress,u16 ARPEntryOffset)
1271 {
1272  u32 lEntryAddress;

1274  u64 lMACAddress;
1275 
1276  lEntryAddress = ARPEntryOffset<<3;
1277  lEntryAddress&=0x00000FF8;
1278  lDataLow=XGMAC_ReadReg(BaseAddress,lEntryAddress);
1279  /* Set the last bit*/

1282  /* Process the 48 bit data*/

1285  lMACAddress |= (u64)(lDataLow&0x00000000FFFFFFFF);
1286  return (lMACAddress&0x0000FFFFFFFFFFFF);
